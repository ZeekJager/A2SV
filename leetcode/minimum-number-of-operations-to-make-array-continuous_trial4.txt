class Solution:
    def minOperations(self, nums: List[int]) -> int:
        n = len(nums)
        arr = sorted(set(nums))
        m = len(arr)

        ans = n
        left = 0

        for right in range(m):
            while arr[right] - arr[left] > n - 1:
                left += 1

            window_size = right - left + 1
            ans = min(ans, n - window_size)

        return ans
