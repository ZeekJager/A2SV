class Solution:
    def pancakeSort(self, arr: List[int]) -> List[int]:
        res = []
        n = len(arr)
        for i in range(n, 1, -1):
            idx=arr.index(i)

            if idx==i-1:
                continue
            if idx!=0:
                arr[:idx+1] = reversed(arr[:idx+1])
                res.append((idx + 1))
            
            arr[:i] = reversed(arr[:i])
            res.append(i)

        return res
