class Solution:
    def isPalindrome(self, s: str) -> bool:
        s=''.join(ch for ch in s if ch.isalnum())
        s=s.lower()
        n=len(s)
        i=0
        j=n-1
        while i<j:
            if s[i]== s[j]:
                i+=1
                j-=1
            else:
                return False
        return True
