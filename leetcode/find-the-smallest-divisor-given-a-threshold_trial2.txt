class Solution:
    def smallestDivisor(self, nums: list[int], threshold: int) -> int:

        left, right = 1, max(nums)
        
        while left < right:
            mid = (left + right) // 2

            current_sum = sum((num + mid - 1) // mid for num in nums) 
            if current_sum <= threshold:
                right = mid
            else:
                left = mid + 1
                
        return left
