class Solution:
    def smallestNumber(self, num: int) -> int:
        if num == 0:
            return 0
        
        digits = list(str(abs(num)))
        
        if num > 0:
            digits.sort()
            # Move first non-zero to the front
            for i, d in enumerate(digits):
                if d != '0':
                    digits[0], digits[i] = digits[i], digits[0]
                    break
            return int(''.join(digits))
        else:
            digits.sort(reverse=True)
            return -int(''.join(digits))
