class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:

        prefix = 0
        count = 0
        freq = defaultdict(int)
        freq[0] = 1
        
        for num in nums:
            prefix += num % 2   # 1 if odd, 0 if even
            
            if prefix - k in freq:
                count += freq[prefix - k]
            
            freq[prefix] += 1
        
        return count
