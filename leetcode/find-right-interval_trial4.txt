class Solution:
    def findRightInterval(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
    
        starts = sorted((intervals[i][0], i) for i in range(n))
        start_values = [s[0] for s in starts]
        
        result = []
        
        for i in range(n):
            end = intervals[i][1]
            
            idx = bisect_left(start_values, end)
            
            if idx == n:
                result.append(-1)
            else:
                result.append(starts[idx][1])
        
        return result
